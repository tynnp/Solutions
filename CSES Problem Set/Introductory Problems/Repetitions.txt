</> Tóm tắt yêu cầu
    - Cho một chuỗi có độ dài n gồm các ký tự A, C, G, T.
    - Tìm chuỗi con chứa một ký tự liên tiếp dài nhất.

</> Cách giải
    - Cho biến cnt = ans = 1.
    - Duyệt qua chuỗi s từ chỉ số i = 1 đến n-1.
        + Nếu s[i-1] == s[i] tức là 2 vị trí liên tiếp có cùng ký tự thì cnt += 1.
        + nếu s[i-1] != s[i] thì cnt = 1.
        + Ở mỗi bước cập nhật ans = max(ans, cnt).
    - Kết quả cần tìm là ans.

</> Code C++
    #include <iostream>
    using namespace std;

    int main() {
        ios_base::sync_with_stdio(0);
        cin.tie(0);
        
        string s;
        int cnt = 1, ans = 1;
        
        cin >> s;
        for (int i = 1; i < s.size(); i++) {
            if (s[i-1] == s[i]) cnt++;
            else cnt = 1;
            ans = max(ans, cnt);
        }
        
        cout << ans;
        return 0;
    }

</> Code Python
    s = input()
    cnt = ans = 1

    for i in range(1, len(s)):
        if s[i-1] == s[i]:
            cnt += 1
        else:
            cnt = 1 
        ans = max(ans, cnt)

    print(ans)
