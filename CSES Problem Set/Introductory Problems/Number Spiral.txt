</> Tóm tắt yêu cầu
    - Cho xoắn ốc như hình.
    - Có t bộ test, mỗi bộ test đưa ra giá trị tại ô của hàng y cột x.

</> Cách giải
    - Tạm gọi ô tại hàng y cột x là (y, x) có giá trị là ans.
    - Xét (3, 1) = 5 && (3, 2) = 6 && (3, 3) = 7 && (5, 1) = 17 && (5, 2) = 18 && ... -> khi y >= x và y lẻ -> ans = y^2 - 2y + x + 1
    - Xét (4, 1) = 16 && (4, 2) = 15 && (4, 3) = 14 && (2, 1) = 4 && (2, 2) = 3 && ... -> khi y >= x và y chẵn -> ans = y^2 - x + 1
    - Xét (1, 5) = 25 && (2, 5) = 24 && (3, 5) = 23 && (2, 3) = 8 && (3, 3) = 7 && ... -> khi x >= y và x lẻ -> ans = x^2 - y + 1
    - Xét (1, 2) = 2 && (2, 2) = 3 && (1, 4) = 10 && (2, 4) = 11 && (2, 5) = 12 && ... -> khi x >= y và x chẵn -> ans = x^2 - 2x + y + 1
    - Giới hạn y và x đến 10^9 -> lưu ý tràn số khi sử dụng kiểu int trong C++.

</> Code C++
    #include <iostream>
    using namespace std;

    int main() {
        ios_base::sync_with_stdio(0);
        cin.tie(0);
        
        long long t, y, x, ans;
        cin >> t;
        
        while (t--) {
            cin >> y >> x;
            
            if (y >= x) {
                if (y & 1) ans = y*y - 2*y + x + 1;
                else ans = y*y - x + 1;
            } else {
                if (x & 1) ans = x*x - y + 1;
                else ans = x*x - 2*x + y + 1;
            }
            
            cout << ans << '\n';
        }
        
        return 0;
    }
