</> Tóm tắt yêu cầu
    - Cho số nguyên dương n.
    - Sắp xếp dãy từ 1 đến n sao cho không có phần tử liền kề hơn kém nhau 1 đơn vị.
    - Nếu không có cách sắp xếp thì in ra "NO SOLUTION".

</> Cách giải
    - Đơn giản nhất là in số lẻ và số chẵn.
    - Trường hợp n bằng 2 hoặc 3 là bất khả thi => NO SOLUTION.
    - Trường hợp từ n = 4:
        + Nếu in 1 3 2 4 -> 3 và 2 hơn kém nhau 1 đơn vị -> sai.
        + Nếu in 2 4 1 3 -> OK -> in số chẵn trước rồi số lẻ sau.
    - Lưu ý trường hợp n = 1 vẫn được.

</> Code C++
    #include <iostream>
    #include <vector>
    using namespace std;

    int main() {
        ios_base::sync_with_stdio(0);
        cin.tie(0);
        
        int n;
        vector<int> a, b;
        cin >> n;
        
        if (n == 2 || n == 3) {
            cout << "NO SOLUTION";
            return 0;
        }
        
        for (int i = 1; i <= n; i++) {
            if (i & 1) a.push_back(i);
            else b.push_back(i);
        }
        
        for (int x : b) cout << x << " ";
        for (int x : a) cout << x << " ";
        
        return 0;
    }

</> Code Python
    n = int(input())
    a = []
    b = []

    if n == 2 or n == 3:
        print("NO SOLUTION")
        exit()

    for i in range(1, n+1):
        if i & 1:
            a.append(i)
        else:
            b.append(i)

    for x in b:
        print(x, end=" ")
    for x in a:
        print(x, end=" ")
